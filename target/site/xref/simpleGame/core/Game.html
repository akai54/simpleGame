1   package simpleGame.core;
2   
3   import java.util.Random;
4   
5   import simpleGame.exception.OutOfBoardException;
6   
7   /**
8    * Class that allows to displayer the board and control the game.
9    * @author Erwan Bousse
10    */
11  public class Game {
12  
13      /**
14       * The board of the game
15       */
16      private Board board;
17  
18      /**
19       * Constructs a Game with 2 pawns on a 4x4 board.
20       */
21      public Game() {
22          board = new Board(2,4,4, new Random().nextInt(4), new Random().nextInt(4));
23      }
24  
25      /**
26       * The game is over if there is only one pawn left or if
27       * a pawn possesses 5 gold or more.
28       * @return true if the game is over
29       */
30      public boolean isGameOver() {
31          return (board.numberOfPawns()==1)
32                 || (board.maxGold() >= 3);
33      }
34  
35      /**
36       * To present the state of the game: the board and is the game is over or not.
37       * @return A string that describes the current state of the game.
38       */
39      public String toString() {
40          String result = board.toString();
41          if (isGameOver())
42              result +="\n\n Game over";
43          return result;
44      }
45  
46      /**
47       * Moves the current pawn in a given direction.
48       * @param d The direction in which the pawn must move.
49       * @return A message regarding what happened.
50       * @throws OutOfBoardException If the move is invalid regarding the board.
51       */
52      public String moveNextPawn(Direction d) throws
53          OutOfBoardException {
54          return this.board.getNextPawn().move(d);
55      }
56  
57  }
