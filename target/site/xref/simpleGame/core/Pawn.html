57      /**
58       * Creates a Pawn with 2 hitpoints and 0 gold.
59       * @param n The letter that represents the pawn.
60       * @param x Position on the x axis
61       * @param y Position on the y axis
62       * @param board The board on which the pawn is located
63       */
64      public Pawn(char n, int x, int y, Board board) {
65          this.letter = n;
66          this.x = x;
67          this.y = y;
68          this.board = board;
69          this.hitpoints = 2;
70          this.gold = 0;
71      }
72  
73  
74      /**
75       * Moves the pawn in a direction.
76       * @param d The direction to move the pawn to.
77       * @return A message that explain what happened during the movement.
78       * @throws OutOfBoardException If the pawn tries to move out of the board.
79       */
80      public String move(Direction d) throws
81          OutOfBoardException {
82  
83          String message = "";
84  
85          int newx = x;
86          int newy = y;
87  
88          switch(d) {
89          case Up:
90              newy++ ;
91              break;
92          case Down:
93              newy-- ;
94              break;
95          case Left:
96              newx-- ;
97              break;
98          case Right:
99              newx++ ;
100             break;
101         }
102 
103         if 	(newy <= board.getYSize()
104                 && newx <= board.getXSize()
105                 && newy > 0
106                 && newx > 0) {
107             Pawn content = board.getSquareContent(newx,newy);
108             if (content == null) {
109                 x = newx;
110                 y = newy;
111             }
112             else {
113                 message = this.attack(content);
114             }
115         } else {
116             throw new OutOfBoardException(newx,newy);
117         }
118 
119         return message;
120     }
121 
122     /**
123      * Makes the pawn attack another pawn.
124      * The enemy pawn should suffer 1 damage,
125      * but it should suffer 2 damages if the current pawn
126      * is on a bonus square.
127      * @param ennemy The attacked pawn.
128      * @return A message that explain what happened during the assault.
129      */
130     private String attack(Pawn enemy) {
131         String message=this.letter + " attacks!\n";
132         if (this.board.isBonusSquare(x, y))
133             message+=enemy.suffer(2);
134         else
135             message+=enemy.suffer(1);
136         if (enemy.isDead()) gold++;
137         return message;
138     }
139 
140     /**
141      * To make the Pawn lose hitpoints.
142      * If the pawn reaches 0 hitpoints, it is removed
143      * from the board.
144      * @param i The number of hitpoints to lose.
145      * @return A message that explain what happened with the wounds.
146      */
147     private String suffer(int i) {
148         String message = this.letter+" loses "+i
149                          +" hitpoints.";
150         hitpoints = hitpoints - i;
151         if (hitpoints <= 0) {
152             this.board.removePawn(this);
153             message += this.letter+" is dead.";
154         }
155         return message;
156     }
157     
158     /**
159      * To know whether a pawn is dead or not.
160      * @return True if the pawn is dead, false otherwise.
161      */
162     public boolean isDead() {
163     	return this.hitpoints == 0;
164     }
